		const: [type];

	}

		single: EXP_RIGHT;
		multi: squareL VALUELIST squareR;
	}
	VALUELIST: EXP_RIGHT VALUES ;
	VALUES: NEXTVALUE*;
	NEXTVALUE: comma EXP_RIGHT;
